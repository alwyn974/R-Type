34 /*
35 Converts a 128-bits unsigned int to an UUIDv4 string representation.
36 Uses SIMD via Intel's AVX2 instruction set.
37 */
38 void inline m128itos(__m128i x, char* mem) {
39 // Expand each byte in x to two bytes in res
40 // i.e. 0x12345678 -> 0x0102030405060708
41 // Then translate each byte to its hex ascii representation
42 // i.e. 0x0102030405060708 -> 0x3132333435363738
43 const __m256i mask = _mm256_set1_epi8(0x0F);
44 const __m256i add = _mm256_set1_epi8(0x06);
45 const __m256i alpha_mask = _mm256_set1_epi8(0x10);
46 const __m256i alpha_offset = _mm256_set1_epi8(0x57);
47
48 __m256i a = _mm256_castsi128_si256(x);
49 __m256i as = _mm256_srli_epi64(a, 4);
50 __m256i lo = _mm256_unpacklo_epi8(as, a);
51 __m128i hi = _mm256_castsi256_si128(_mm256_unpackhi_epi8(as, a));
52 __m256i c = _mm256_inserti128_si256(lo, hi, 1);
53 __m256i d = _mm256_and_si256(c, mask);
54 __m256i alpha = _mm256_slli_epi64(_mm256_and_si256(_mm256_add_epi8(d, add), alpha_mask), 3);
55 __m256i offset = _mm256_blendv_epi8(_mm256_slli_epi64(add, 3), alpha_offset, alpha);
56 __m256i res = _mm256_add_epi8(d, offset);
57
58 // Add dashes between blocks as specified in RFC-4122
59 // 8-4-4-4-12
60 const __m256i dash_shuffle = _mm256_set_epi32(0x0b0a0908, 0x07060504, 0x80030201, 0x00808080, 0x0d0c800b, 0x0a090880, 0x07060504, 0x03020100);
61 const __m256i dash = _mm256_set_epi64x(0x0000000000000000ull, 0x2d000000002d0000ull, 0x00002d000000002d, 0x0000000000000000ull);
62
63 __m256i resd = _mm256_shuffle_epi8(res, dash_shuffle);
64 resd = _mm256_or_si256(resd, dash);
65
66 _mm256_storeu_si256((__m256i*)mem, betole256(resd));
67 *(uint16_t*)(mem+16) = betole16(_mm256_extract_epi16(res, 7));
68 *(uint32_t*)(mem+32) = betole32(_mm256_extract_epi32(res, 7));
69 }

71 /*
72 Converts an UUIDv4 string representation to a 128-bits unsigned int.
73 Uses SIMD via Intel's AVX2 instruction set.
74 */
75 __m128i inline stom128i(const char* mem) {
76 // Remove dashes and pack hex ascii bytes in a 256-bits int
77 const __m256i dash_shuffle = _mm256_set_epi32(0x80808080, 0x0f0e0d0c, 0x0b0a0908, 0x06050403, 0x80800f0e, 0x0c0b0a09, 0x07060504, 0x03020100);
78
79 __m256i x = betole256(_mm256_loadu_si256((__m256i*)mem));
80 x = _mm256_shuffle_epi8(x, dash_shuffle);
81 x = _mm256_insert_epi16(x, betole16(*(uint16_t*)(mem+16)), 7);
82 x = _mm256_insert_epi32(x, betole32(*(uint32_t*)(mem+32)), 7);
83
84 // Build a mask to apply a different offset to alphas and digits
85 const __m256i sub = _mm256_set1_epi8(0x2F);
86 const __m256i mask = _mm256_set1_epi8(0x20);
87 const __m256i alpha_offset = _mm256_set1_epi8(0x28);
88 const __m256i digits_offset = _mm256_set1_epi8(0x01);
89 const __m256i unweave = _mm256_set_epi32(0x0f0d0b09, 0x0e0c0a08, 0x07050301, 0x06040200, 0x0f0d0b09, 0x0e0c0a08, 0x07050301, 0x06040200);
90 const __m256i shift = _mm256_set_epi32(0x00000000, 0x00000004, 0x00000000, 0x00000004, 0x00000000, 0x00000004, 0x00000000, 0x00000004);
91
92 // Translate ascii bytes to their value
93 // i.e. 0x3132333435363738 -> 0x0102030405060708
94 // Shift hi-digits
95 // i.e. 0x0102030405060708 -> 0x1002300450067008
96 // Horizontal add
97 // i.e. 0x1002300450067008 -> 0x12345678
98 __m256i a = _mm256_sub_epi8(x, sub);
99 __m256i alpha = _mm256_slli_epi64(_mm256_and_si256(a, mask), 2);
100 __m256i sub_mask = _mm256_blendv_epi8(digits_offset, alpha_offset, alpha);
101 a = _mm256_sub_epi8(a, sub_mask);
102 a = _mm256_shuffle_epi8(a, unweave);
103 a = _mm256_sllv_epi32(a, shift);
104 a = _mm256_hadd_epi32(a, _mm256_setzero_si256());
105 a = _mm256_permute4x64_epi64(a, 0b00001000);
106
107 return _mm256_castsi256_si128(a);
108 }

157 UUID& operator=(const UUID &other) {

166 friend bool operator==(const UUID &lhs, const UUID &rhs) {
167 __m128i x = _mm_load_si128((__m128i*)lhs.data);
168 __m128i y = _mm_load_si128((__m128i*)rhs.data);
169
170 __m128i neq = _mm_xor_si128(x, y);
171 return _mm_test_all_zeros(neq, neq);
172 }

174 friend bool operator<(const UUID &lhs, const UUID &rhs) {
175 // There are no trivial 128-bits comparisons in SSE/AVX
176 // It's faster to compare two uint64_t
177 uint64_t *x = (uint64_t*)lhs.data;
178 uint64_t *y = (uint64_t*)rhs.data;
179 return *x < *y || (*x == *y && *(x + 1) < *(y + 1));
180 }

240 /*
241 Generates UUIDv4 from a provided random generator (c++11 <random> module)
242 std::mt19937_64 is highly recommended as it has a SIMD implementation that
243 makes it very fast and it produces high quality randomness.
244 */
245 template <typename RNG>
246 class UUIDGenerator {
247 public:
248 UUIDGenerator() : generator(new RNG(std::random_device()())), distribution(std::numeric_limits<uint64_t>::min(), std::numeric_limits<uint64_t>::max())
249 {}
250
251 UUIDGenerator(uint64_t seed) : generator(new RNG(seed)), distribution(std::numeric_limits<uint64_t>::min(), std::numeric_limits<uint64_t>::max())
252 {}
253
254 UUIDGenerator(RNG &gen) : generator(gen), distribution(std::numeric_limits<uint64_t>::min(), std::numeric_limits<uint64_t>::max())
255 {}
256
257 /* Generates a new UUID */
258 UUID getUUID() {
259 // The two masks set the uuid version (4) and variant (1)
260 const __m128i and_mask = _mm_set_epi64x(0xFFFFFFFFFFFFFF3Full, 0xFF0FFFFFFFFFFFFFull);
261 const __m128i or_mask = _mm_set_epi64x(0x0000000000000080ull, 0x0040000000000000ull);
262 __m128i n = _mm_set_epi64x(distribution(*generator), distribution(*generator));
263 __m128i uuid = _mm_or_si128(_mm_and_si128(n, and_mask), or_mask);
264
265 return UUID(uuid);
266 }
267
268 private:
269 std::shared_ptr<RNG> generator;
270 std::uniform_int_distribution<uint64_t> distribution;
271 };
